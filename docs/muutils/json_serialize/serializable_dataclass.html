 83def SerializableDataclass__validate_field_type(
 84    self: SerializableDataclass,
 85    field: SerializableField | str,
 86    on_typecheck_error: ErrorMode = _DEFAULT_ON_TYPECHECK_ERROR,
 87) -> bool:
 88    """given a dataclass, check the field matches the type hint
 89
 90    # Parameters:
 91     - `self : SerializableDataclass`
 92       `SerializableDataclass` instance
 93     - `field : SerializableField | str`
 94        field to validate, will get from `self.__dataclass_fields__` if an `str`
 95     - `on_typecheck_error : ErrorMode`
 96        what to do if type checking throws an exception (except, warn, ignore). If `ignore` and an exception is thrown, the function will return `False`
 97       (defaults to `_DEFAULT_ON_TYPECHECK_ERROR`)
 98
 99    # Returns:
100     - `bool`
101        if the field type is correct. `False` if the field type is incorrect or an exception is thrown and `on_typecheck_error` is `ignore`
102    """
103    on_typecheck_error = ErrorMode.from_any(on_typecheck_error)
104
105    # get field
106    _field: SerializableField
107    if isinstance(field, str):
108        _field = self.__dataclass_fields__[field]  # type: ignore[attr-defined]
109    else:
110        _field = field
111
112    # do nothing case
113    if not _field.assert_type:
114        return True
115
116    # if field is not `init` or not `serialize`, skip but warn
117    # TODO: how to handle fields which are not `init` or `serialize`?
118    if not _field.init or not _field.serialize:
119        warnings.warn(
120            f"Field '{_field.name}' on class {self.__class__} is not `init` or `serialize`, so will not be type checked",
121            FieldIsNotInitOrSerializeWarning,
122        )
123        return True
124
125    assert isinstance(
126        _field, SerializableField
127    ), f"Field '{_field.name = }' on class {self.__class__ = } is not a SerializableField, but a {type(_field) = }"
128
129    # get field type hints
130    try:
131        field_type_hint: Any = get_cls_type_hints(self.__class__)[_field.name]
132    except KeyError as e:
133        on_typecheck_error.process(
134            (
135                f"Cannot get type hints for {self.__class__.__name__}, field {_field.name = } and so cannot validate.\n"
136                + f"{get_cls_type_hints(self.__class__) = }\n"
137                + f"Python version is {sys.version_info = }. You can:\n"
138                + f"  - disable `assert_type`. Currently: {_field.assert_type = }\n"
139                + f"  - use hints like `typing.Dict` instead of `dict` in type hints (this is required on python 3.8.x). You had {_field.type = }\n"
140                + "  - use python 3.9.x or higher\n"
141                + "  - specify custom type validation function via `custom_typecheck_fn`\n"
142            ),
143            except_cls=TypeError,
144            except_from=e,
145        )
146        return False
147
148    # get the value
149    value: Any = getattr(self, _field.name)
150
151    # validate the type
152    try:
153        type_is_valid: bool
154        # validate the type with the default type validator
155        if _field.custom_typecheck_fn is None:
156            type_is_valid = validate_type(value, field_type_hint)
157        # validate the type with a custom type validator
158        else:
159            type_is_valid = _field.custom_typecheck_fn(field_type_hint)
160
161        return type_is_valid
162
163    except Exception as e:
164        on_typecheck_error.process(
165            "exception while validating type: "
166            + f"{_field.name = }, {field_type_hint = }, {type(field_type_hint) = }, {value = }",
167            except_cls=ValueError,
168            except_from=e,
169        )
170        return False

173def SerializableDataclass__validate_fields_types__dict(
174    self: SerializableDataclass,
175    on_typecheck_error: ErrorMode = _DEFAULT_ON_TYPECHECK_ERROR,
176) -> dict[str, bool]:
177    """validate the types of all the fields on a SerializableDataclass. calls `SerializableDataclass__validate_field_type` for each field
178
179    returns a dict of field names to bools, where the bool is if the field type is valid
180    """
181    on_typecheck_error = ErrorMode.from_any(on_typecheck_error)
182
183    # if except, bundle the exceptions
184    results: dict[str, bool] = dict()
185    exceptions: dict[str, Exception] = dict()
186
187    # for each field in the class
188    cls_fields: typing.Sequence[SerializableField] = dataclasses.fields(self)  # type: ignore[arg-type, assignment]
189    for field in cls_fields:
190        try:
191            results[field.name] = self.validate_field_type(field, on_typecheck_error)
192        except Exception as e:
193            results[field.name] = False
194            exceptions[field.name] = e
195
196    # figure out what to do with the exceptions
197    if len(exceptions) > 0:
198        on_typecheck_error.process(
199            f"Exceptions while validating types of fields on {self.__class__.__name__}: {[x.name for x in cls_fields]}"
200            + "\n\t"
201            + "\n\t".join([f"{k}:\t{v}" for k, v in exceptions.items()]),
202            except_cls=ValueError,
203            # HACK: ExceptionGroup not supported in py < 3.11, so get a random exception from the dict
204            except_from=list(exceptions.values())[0],
205        )
206
207    return results

259    def diff(
260        self, other: "SerializableDataclass", of_serialized: bool = False
261    ) -> dict[str, Any]:
262        """get a rich and recursive diff between two instances of a serializable dataclass
263
264        ```python
265        >>> Myclass(a=1, b=2).diff(Myclass(a=1, b=3))
266        {'b': {'self': 2, 'other': 3}}
267        >>> NestedClass(x="q1", y=Myclass(a=1, b=2)).diff(NestedClass(x="q2", y=Myclass(a=1, b=3)))
268        {'x': {'self': 'q1', 'other': 'q2'}, 'y': {'b': {'self': 2, 'other': 3}}}
269        ```
270
271        # Parameters:
272         - `other : SerializableDataclass`
273           other instance to compare against
274         - `of_serialized : bool`
275           if true, compare serialized data and not raw values
276           (defaults to `False`)
277
278        # Returns:
279         - `dict[str, Any]`
280
281
282        # Raises:
283         - `ValueError` : if the instances are not of the same type
284         - `ValueError` : if the instances are `dataclasses.dataclass` but not `SerializableDataclass`
285        """
286        # match types
287        if type(self) != type(other):
288            raise ValueError(
289                f"Instances must be of the same type, but got {type(self) = } and {type(other) = }"
290            )
291
292        # initialize the diff result
293        diff_result: dict = {}
294
295        # if they are the same, return the empty diff
296        if self == other:
297            return diff_result
298
299        # if we are working with serialized data, serialize the instances
300        if of_serialized:
301            ser_self: dict = self.serialize()
302            ser_other: dict = other.serialize()
303
304        # for each field in the class
305        for field in dataclasses.fields(self):  # type: ignore[arg-type]
306            # skip fields that are not for comparison
307            if not field.compare:
308                continue
309
310            # get values
311            field_name: str = field.name
312            self_value = getattr(self, field_name)
313            other_value = getattr(other, field_name)
314
315            # if the values are both serializable dataclasses, recurse
316            if isinstance(self_value, SerializableDataclass) and isinstance(
317                other_value, SerializableDataclass
318            ):
319                nested_diff: dict = self_value.diff(
320                    other_value, of_serialized=of_serialized
321                )
322                if nested_diff:
323                    diff_result[field_name] = nested_diff
324            # only support serializable dataclasses
325            elif dataclasses.is_dataclass(self_value) and dataclasses.is_dataclass(
326                other_value
327            ):
328                raise ValueError("Non-serializable dataclass is not supported")
329            else:
330                # get the values of either the serialized or the actual values
331                self_value_s = ser_self[field_name] if of_serialized else self_value
332                other_value_s = ser_other[field_name] if of_serialized else other_value
333                # compare the values
334                if not array_safe_eq(self_value_s, other_value_s):
335                    diff_result[field_name] = {"self": self_value, "other": other_value}
336
337        # return the diff result
338        return diff_result
339
340    def update_from_nested_dict(self, nested_dict: dict[str, Any]):
341        """update the instance from a nested dict, useful for configuration from command line args
342
343        # Parameters:
344            - `nested_dict : dict[str, Any]`
345                nested dict to update the instance with
346        """
347        for field in dataclasses.fields(self):  # type: ignore[arg-type]
348            field_name: str = field.name
349            self_value = getattr(self, field_name)
350
351            if field_name in nested_dict:
352                if isinstance(self_value, SerializableDataclass):
353                    self_value.update_from_nested_dict(nested_dict[field_name])
354                else:
355                    setattr(self, field_name, nested_dict[field_name])

358        "deep copy by serializing and loading the instance to json"
359        return self.__class__.load(json.loads(json.dumps(self.serialize())))

361    def __deepcopy__(self, memo: dict) -> "SerializableDataclass":

394# Step 3: Create a custom serializable_dataclass decorator
395def serializable_dataclass(
396    # this should be `_cls: Type[T] | None = None,` but mypy doesn't like it
397    _cls=None,  # type: ignore
398    *,
399    init: bool = True,
400    repr: bool = True,  # this overrides the actual `repr` builtin, but we have to match the interface of `dataclasses.dataclass`
401    eq: bool = True,
402    order: bool = False,
403    unsafe_hash: bool = False,
404    frozen: bool = False,
405    properties_to_serialize: Optional[list[str]] = None,
406    register_handler: bool = True,
407    on_typecheck_error: ErrorMode = _DEFAULT_ON_TYPECHECK_ERROR,
408    on_typecheck_mismatch: ErrorMode = _DEFAULT_ON_TYPECHECK_MISMATCH,
409    **kwargs,
410):
411    """decorator to make a dataclass serializable. must also make it inherit from `SerializableDataclass`
412
413    types will be validated (like pydantic) unless `on_typecheck_mismatch` is set to `ErrorMode.IGNORE`
414
415    behavior of most kwargs matches that of `dataclasses.dataclass`, but with some additional kwargs
416
417    Returns the same class as was passed in, with dunder methods added based on the fields defined in the class.
418
419    Examines PEP 526 __annotations__ to determine fields.
420
421    If init is true, an __init__() method is added to the class. If repr is true, a __repr__() method is added. If order is true, rich comparison dunder methods are added. If unsafe_hash is true, a __hash__() method function is added. If frozen is true, fields may not be assigned to after instance creation.
422
423    ```python
424    @serializable_dataclass(kw_only=True)
425    class Myclass(SerializableDataclass):
426        a: int
427        b: str
428    ```
429    ```python
430    >>> Myclass(a=1, b="q").serialize()
431    {'__format__': 'Myclass(SerializableDataclass)', 'a': 1, 'b': 'q'}
432    ```
433
434    # Parameters:
435     - `_cls : _type_`
436       class to decorate. don't pass this arg, just use this as a decorator
437       (defaults to `None`)
438     - `init : bool`
439       (defaults to `True`)
440     - `repr : bool`
441       (defaults to `True`)
442     - `order : bool`
443       (defaults to `False`)
444     - `unsafe_hash : bool`
445       (defaults to `False`)
446     - `frozen : bool`
447       (defaults to `False`)
448     - `properties_to_serialize : Optional[list[str]]`
449       **SerializableDataclass only:** which properties to add to the serialized data dict
450       (defaults to `None`)
451     - `register_handler : bool`
452        **SerializableDataclass only:** if true, register the class with ZANJ for loading
453       (defaults to `True`)
454     - `on_typecheck_error : ErrorMode`
455        **SerializableDataclass only:** what to do if type checking throws an exception (except, warn, ignore). If `ignore` and an exception is thrown, type validation will still return false
456     - `on_typecheck_mismatch : ErrorMode`
457        **SerializableDataclass only:** what to do if a type mismatch is found (except, warn, ignore). If `ignore`, type validation will return `True`
458
459    # Returns:
460     - `_type_`
461       _description_
462
463    # Raises:
464     - `ValueError` : _description_
465     - `ValueError` : _description_
466     - `ValueError` : _description_
467     - `AttributeError` : _description_
468     - `ValueError` : _description_
469    """
470    # -> Union[Callable[[Type[T]], Type[T]], Type[T]]:
471    on_typecheck_error = ErrorMode.from_any(on_typecheck_error)
472    on_typecheck_mismatch = ErrorMode.from_any(on_typecheck_mismatch)
473
474    if properties_to_serialize is None:
475        _properties_to_serialize: list = list()
476    else:
477        _properties_to_serialize = properties_to_serialize
478
479    def wrap(cls: Type[T]) -> Type[T]:
480        # Modify the __annotations__ dictionary to replace regular fields with SerializableField
481        for field_name, field_type in cls.__annotations__.items():
482            field_value = getattr(cls, field_name, None)
483            if not isinstance(field_value, SerializableField):
484                if isinstance(field_value, dataclasses.Field):
485                    # Convert the field to a SerializableField while preserving properties
486                    field_value = SerializableField.from_Field(field_value)
487                else:
488                    # Create a new SerializableField
489                    field_value = serializable_field()
490                setattr(cls, field_name, field_value)
491
492        # special check, kw_only is not supported in python <3.9 and `dataclasses.MISSING` is truthy
493        if sys.version_info < (3, 10):
494            if "kw_only" in kwargs:
495                if kwargs["kw_only"] == True:  # noqa: E712
496                    raise ValueError("kw_only is not supported in python >=3.9")
497                else:
498                    del kwargs["kw_only"]
499
500        # call `dataclasses.dataclass` to set some stuff up
501        cls = dataclasses.dataclass(  # type: ignore[call-overload]
502            cls,
503            init=init,
504            repr=repr,
505            eq=eq,
506            order=order,
507            unsafe_hash=unsafe_hash,
508            frozen=frozen,
509            **kwargs,
510        )
511
512        # copy these to the class
513        cls._properties_to_serialize = _properties_to_serialize.copy()  # type: ignore[attr-defined]
514
515        # ======================================================================
516        # define `serialize` func
517        # done locally since it depends on args to the decorator
518        # ======================================================================
519        def serialize(self) -> dict[str, Any]:
520            result: dict[str, Any] = {
521                "__format__": f"{self.__class__.__name__}(SerializableDataclass)"
522            }
523            # for each field in the class
524            for field in dataclasses.fields(self):  # type: ignore[arg-type]
525                # need it to be our special SerializableField
526                if not isinstance(field, SerializableField):
527                    raise ValueError(
528                        f"Field '{field.name}' on class {self.__class__.__module__}.{self.__class__.__name__} is not a SerializableField, "
529                        f"but a {type(field)} "
530                        "this state should be inaccessible, please report this bug!"
531                    )
532
533                # try to save it
534                if field.serialize:
535                    try:
536                        # get the val
537                        value = getattr(self, field.name)
538                        # if it is a serializable dataclass, serialize it
539                        if isinstance(value, SerializableDataclass):
540                            value = value.serialize()
541                        # if the value has a serialization function, use that
542                        if hasattr(value, "serialize") and callable(value.serialize):
543                            value = value.serialize()
544                        # if the field has a serialization function, use that
545                        # it would be nice to be able to override a class's `.serialize()`, but that could lead to some inconsistencies!
546                        elif field.serialization_fn:
547                            value = field.serialization_fn(value)
548
549                        # store the value in the result
550                        result[field.name] = value
551                    except Exception as e:
552                        raise ValueError(
553                            "\n".join(
554                                [
555                                    f"Error serializing field '{field.name}' on class {self.__class__.__module__}.{self.__class__.__name__}",
556                                    f"{field = }",
557                                    f"{value = }",
558                                    f"{self = }",
559                                ]
560                            )
561                        ) from e
562
563            # store each property if we can get it
564            for prop in self._properties_to_serialize:
565                if hasattr(cls, prop):
566                    value = getattr(self, prop)
567                    result[prop] = value
568                else:
569                    raise AttributeError(
570                        f"Cannot serialize property '{prop}' on class {self.__class__.__module__}.{self.__class__.__name__}"
571                        + f"but it is in {self._properties_to_serialize = }"
572                        + f"\n{self = }"
573                    )
574
575            return result
576
577        # ======================================================================
578        # define `load` func
579        # done locally since it depends on args to the decorator
580        # ======================================================================
581        # mypy thinks this isnt a classmethod
582        @classmethod  # type: ignore[misc]
583        def load(cls, data: dict[str, Any] | T) -> Type[T]:
584            # HACK: this is kind of ugly, but it fixes a lot of issues for when we do recursive loading with ZANJ
585            if isinstance(data, cls):
586                return data
587
588            assert isinstance(
589                data, typing.Mapping
590            ), f"When loading {cls.__name__ = } expected a Mapping, but got {type(data) = }:\n{data = }"
591
592            cls_type_hints: dict[str, Any] = get_cls_type_hints(cls)
593
594            # initialize dict for keeping what we will pass to the constructor
595            ctor_kwargs: dict[str, Any] = dict()
596
597            # iterate over the fields of the class
598            for field in dataclasses.fields(cls):
599                # check if the field is a SerializableField
600                assert isinstance(
601                    field, SerializableField
602                ), f"Field '{field.name}' on class {cls.__name__} is not a SerializableField, but a {type(field)}. this state should be inaccessible, please report this bug!\nhttps://github.com/mivanit/muutils/issues/new"
603
