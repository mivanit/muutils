154def load_array(arr: JSONitem, array_mode: Optional[ArrayMode] = None) -> Any:
155    """load a json-serialized array, infer the mode if not specified"""
156    # return arr if its already a numpy array
157    if isinstance(arr, np.ndarray) and array_mode is None:
158        return arr
159
160    # try to infer the array_mode
161    array_mode_inferred: ArrayMode = infer_array_mode(arr)
162    if array_mode is None:
163        array_mode = array_mode_inferred
164    elif array_mode != array_mode_inferred:
165        warnings.warn(
166            f"array_mode {array_mode} does not match inferred array_mode {array_mode_inferred}"
167        )
168
169    # actually load the array
170    if array_mode == "array_list_meta":
171        assert isinstance(
172            arr, typing.Mapping
173        ), f"invalid list format: {type(arr) = }\n{arr = }"
174
175        data = np.array(arr["data"], dtype=arr["dtype"])
176        if tuple(arr["shape"]) != tuple(data.shape):
177            raise ValueError(f"invalid shape: {arr}")
178        return data
179
180    elif array_mode == "array_hex_meta":
181        assert isinstance(
182            arr, typing.Mapping
183        ), f"invalid list format: {type(arr) = }\n{arr = }"
184
185        data = np.frombuffer(bytes.fromhex(arr["data"]), dtype=arr["dtype"])
186        return data.reshape(arr["shape"])
187
188    elif array_mode == "array_b64_meta":
189        assert isinstance(
190            arr, typing.Mapping
191        ), f"invalid list format: {type(arr) = }\n{arr = }"
192
193        data = np.frombuffer(base64.b64decode(arr["data"]), dtype=arr["dtype"])
194        return data.reshape(arr["shape"])
195
196    elif array_mode == "list":
197        assert isinstance(
198            arr, typing.Sequence
199        ), f"invalid list format: {type(arr) = }\n{arr = }"
200
201        return np.array(arr)
202    elif array_mode == "external":
203        # assume ZANJ has taken care of it
204        assert isinstance(arr, typing.Mapping)
205        if "data" not in arr:
206            raise KeyError(
207                f"invalid external array, expected key 'data', got keys: '{list(arr.keys())}' and arr: {arr}"
208            )
209        return arr["data"]
210    elif array_mode == "zero_dim":
211        assert isinstance(arr, typing.Mapping)
212        data = np.array(arr["data"])
213        if tuple(arr["shape"]) != tuple(data.shape):
214            raise ValueError(f"invalid shape: {arr}")
215        return data
216    else:
217        raise ValueError(f"invalid array_mode: {array_mode}")
